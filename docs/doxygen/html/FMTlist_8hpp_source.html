132 const boost::dynamic_bitset<> selecinter = selectedbits & baseit->first.getbitsetreference();
133 boost::dynamic_bitset<> reverselect(selecinter);
134 for (size_t loc = thstart; loc < (theme.size() + thstart); ++loc)

136 reverselect[loc] = true;
137 }
138 /*Core::FMTmask testmask(baseit->first);
139 testmask.set(theme, "?");*/

142 /*Core::FMTmask datamask(datait->first);

198 void presolvelist(const FMTmaskfilter& filter,
199 const std::vector<FMTtheme>& originalthemes,
200 const std::vector<FMTtheme>& newthemes)
201 {
202 try {
203 if (!canshrink())
204 {
205 unshrink(originalthemes);
206 }
207 const std::vector<FMTtheme>maskthemes = filter.getselectedthemes(originalthemes);
208 std::vector<std::pair<FMTmask, T>>newdata;
209 for (const std::pair<FMTmask, T>& object : data)
210 {
211 if (filter.canpresolve(object.first,maskthemes))
212 {
213 FMTmask mskkey = object.first;
214 if (!filter.emptyflipped())
215 {
216 mskkey = mskkey.presolve(filter, newthemes);
217 }
218 pushtodata(newdata, mskkey, object.second);
219 }
220 }
221 data.swap(newdata);

355 {
356 std::vector<const T*>allhits;
357 boost::unordered_map<FMTmask, std::vector<int>>::const_iterator fast_it = fastpass.find(newkey);
358 if (fast_it != fastpass.end())
359 {
360 allhits.reserve(fast_it->second.size());
361 for (const int& location : fast_it->second)
362 {
363 allhits.push_back(&data.at(location).second);
364 }
365 }
366 else {
367 fastpass[newkey] = std::vector<int>();
368 int location = 0;
369 for (const std::pair<FMTmask, T>& object : data)
370 {
371 if (newkey.issubsetof(object.first))
372 {
373 fastpass[newkey].push_back(location);
374 allhits.push_back(&object.second);
375 }
376 ++location;
377 }
378 fastpass[newkey].shrink_to_fit();
379 }
