358 std::vector<const T*>allhits;

371 for (const std::pair<FMTmask, T>& object : data)

373 if (newkey.issubsetof(object.first))
374 {
375 fastpass[newkey].push_back(location);
376 allhits.push_back(&object.second);
377 }
378 ++location;
